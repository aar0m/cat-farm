271/// Validates that SingleLinkedList follows a specified ruleset.
272///
273/// - Is a valid Container
274/// - If 'head == nullptr', then 'count' must be '0' and isEmpty() must be true.
275/// - If 'head != nullptr', then 'count' must NOT be '0' and isEmpty() must be false.
276/// - Amount of objects in a SingleLinkedList is equal to 'count'
277///
278/// @return 'true' if SingleLinkedList passes ruleset. 'false' otherwise.
279bool SingleLinkedList::validate() const noexcept {
280 assert( Container::validate() );
281 Node* current = head;
282 size_t finalCount = 0;
283 bool result = true;
284
285 if( !Container::validate() ) {
286 result = false;
287 }
288
289 if( head == nullptr ) {
290 if( ( count != 0 ) || !isEmpty() ) {
291 result = false;
292 }
293 } else {
294 if( ( count == 0 ) || isEmpty() ) {
295 result = false;
296 }
297 }
298
299 while( current != nullptr ) {
300 if( !current->validate() ) {
301 result = false;
302 }
303
304 finalCount += 1;
305 current = current->next;
306 }
307
308 if( finalCount != count ) {
309 result = false;
310 }
311
312 return result;
313}
