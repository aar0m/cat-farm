171/// Validates that Mammal follows a specified ruleset.
172///
173/// - getClassification() must not be empty.
174/// - getColor() must not be empty.
175/// - 'gender' must be valid
176/// - Is a valid Animal
177///
178/// @return 'true' if Mammal passes ruleset. 'false' otherwise.
179bool Mammal::validate() const noexcept{
180 if( ( color.empty() ) ) {
181 cout << PROGRAM_NAME << ": Please set a color for the mammal manually or use generateColor()" << endl;
182 return false;
183 }
184
185 if( getClassification().empty() ) {
186 cout << PROGRAM_NAME << ": Please set a classification for the mammal" << endl;
187 return false;
188 }
189
190 if( !validateGender( gender ) ) {
191 cout << PROGRAM_NAME << ": This animal has an invalid gender." << endl;
192 return false;
193 }
194
195 return Animal::validate();
196}
