80void Name::reset() {
81 if( nameType == FROM_FILE ) {
82 names.clear();
83
84 #ifdef DEBUG
85 cout << "Loading names from [" << fileName << "]: " ;
86 #endif
87
88 ifstream file( fileName, ios::in );
89 if( !file.is_open() ) {
90 throw( runtime_error( "Unable to open [" + fileName + "] for reading" ));
91 }
92
93 string line;
94 while( getline( file, line )) {
95 line = trim( line.data() );
96 if( !Name::validateName( line ) ) { /// As the names are loaded from the file, verify that each name is valid with validateName()
97 continue;
98 }
99 names.push_back( std::move( line )); // Use move semantics
100 }
101
102 #ifdef DEBUG
103 cout << "Loaded " << to_string( names.size() ) << " names" << endl;
104 #endif
105 }
106
107 if( nameType == FROM_SERIAL ) {
108 serial = 0;
109 }
110}

129/// newName should have had ::trim run on it before it's validated here.
130///
131/// The validation rules are:
132///
133/// - The name should not be empty.
134/// - The name should be trimmed for whitespace (both left and right
135/// whitespace should be removed).
136/// - The name should not start with a number or `-`.
137/// - The name should not have any special characters except a `-`. Allowed
138/// characters are `A`-`Z`, `a`-`z`, `0`-`9`, space and `-`.
139/// - The interior whitespace of the name should be trimmed.
140///
141/// If the name is not valid, print an explanation and return `false`
142///
143/// @param newName The name to check
144/// @return `true` if the name is valid. `false` if it's not.
145bool Name::validateName( const std::string_view newName ) {
146
147 if( !validateNotEmpty( newName )) {
148 return false;
149 }
150
151 if( !validateTrimmed( newName )) {
152 return false;
153 }
154
155 if( !validateStartsWithAlpha( newName )) {
156 return false;
157 }
158
159 if( !validateNoSpecialChars( newName )) {
160 return false;
161 }
162
163 // This is a little subtle. At this point, the only "whitespace" allowed is
164 // a space ' ' and a '-', so let's just look for them...
165 if( !validateInteriorWhitespaceTrimmed( newName )) {
166 return false;
167 }
168
169 return true;
170}

230bool Name::validateInteriorWhitespaceTrimmed( const string_view newName ) {
231 string tempName( newName ) ;
232 string trialName { trim_edges( tempName.data() ) }; // Trim whitespace
233
234 for( auto i = trialName.begin() ; i+1 != trialName.end() ; i++ ) {
235 const bool isCurrentWhitespace = *i == ' ' || *i == '-';
236 const bool isNextWhitespace = *( i+1 ) == ' ' || *( i+1 ) == '-';
237
238 if( isCurrentWhitespace && isNextWhitespace ) {
239 /// If not valid, print `The interior whitespace of the name should be trimmed` and return `false`

242 }
